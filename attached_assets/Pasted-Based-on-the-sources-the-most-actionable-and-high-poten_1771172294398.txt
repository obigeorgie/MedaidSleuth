@app.route('/api/query', methods=['GET'])
def query_data():
    """
    Basic filtering: Filter by State and Procedure Code.
    Source [6]: "Allow users to filter by provider, code, state..."
    """
    state = request.args.get('state')
    code = request.args.get('code')
    
    query = "SELECT * FROM claims WHERE 1=1"
    params = []
    
    if state:
        query += " AND state_code = ?"
        params.append(state)
    if code:
        query += " AND procedure_code = ?"
        params.append(code)
        
    query += " ORDER BY month"
    
    df = con.execute(query, params).df()
    # Convert dates to string for JSON serialization
    df['month'] = df['month'].astype(str)
    return jsonify(df.to_dict(orient='records'))

@app.route('/api/scan_fraud', methods=['GET'])
def scan_fraud():
    """
    The 'Secret Sauce': Detects month-over-month billing spikes > 200%.
    Source [2]: "Query for month-over-month growth >200% in claims"
    """
    query = """
    WITH monthly_stats AS (
        SELECT 
            provider_name,
            state_code,
            procedure_code,
            month,
            SUM(total_paid) as monthly_total
        FROM claims
        GROUP BY provider_name, state_code, procedure_code, month
    ),
    lagged_stats AS (
        SELECT 
            *,
            LAG(monthly_total) OVER (PARTITION BY provider_name, procedure_code ORDER BY month) as prev_month_total
        FROM monthly_stats
    )
    SELECT 
        provider_name,
        state_code,
        procedure_code,
        month,
        monthly_total,
        prev_month_total,
        ROUND(((monthly_total - prev_month_total) / prev_month_total) * 100, 2) as growth_percent
    FROM lagged_stats
    WHERE prev_month_total > 0 
      AND ((monthly_total - prev_month_total) / prev_month_total) > 2.0
    ORDER BY growth_percent DESC
    """
    df = con.execute(query).df()
    df['month'] = df['month'].astype(str)
    return jsonify(df.to_dict(orient='records'))
